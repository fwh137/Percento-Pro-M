function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); // 功能 1：獲取所有數據 (改為從 A~L 欄讀取) if (e.parameter.action === "getAllData" || e.parameter.action === "getAssets") { var lastRow = sheet.getLastRow(); var assets = []; // 如果有資料（排除第一列標題的話，lastRow 需 > 1） if (lastRow > 1) { // 讀取第 2 列開始，到最後一列，共 12 欄 (A~L) var rows = sheet.getRange(2, 1, lastRow - 1, 12).getValues(); // 將每列資料對應轉為你的物件格式（請根據你的欄位名稱調整 keys） assets = rows.map(function(row) { return { symbol: row[0], // A欄 qty: row[1], // B欄 price: row[2], // C欄 currency: row[3], // D欄 type: row[4], // E欄 category: row[5], // F欄 // row[6]~row[11] 對應 G~L 欄，需要可繼續加 }; }); } var responseData = { "assets": assets, "hiddenCats": [] }; // 若 hiddenCats 有固定儲存格可再另外讀取 return ContentService.createTextOutput(JSON.stringify(responseData)) .setMimeType(ContentService.MimeType.JSON); } // 功能 2：抓股價功能 (維持不變) var symbol = e.parameter.symbol; if (symbol) { var tempCell = sheet.getRange("Y1"); tempCell.clearContent(); SpreadsheetApp.flush(); tempCell.setFormula('=GOOGLEFINANCE("' + symbol + '", "price")'); SpreadsheetApp.flush(); Utilities.sleep(800); var price = tempCell.getValue(); if (price === "#N/A" || isNaN(price) || price === "") { price = null; } return ContentService.createTextOutput(JSON.stringify({price: price})) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); try { var payload = JSON.parse(e.postData.contents); // 功能：儲存所有數據 (改為寫入 A~L 欄) if (payload.action === "saveAllData" || payload.action === "saveAssets") { var dataToSave = payload.hiddenCats ? payload : payload.data; var assets = dataToSave.assets || []; // 1. 先清空第 2 列以下、A~L 欄的舊資料 var lastRow = sheet.getLastRow(); if (lastRow > 1) { sheet.getRange(2, 1, lastRow - 1, 12).clearContent(); } // 2. 如果有新資料，轉成二維陣列後寫入 if (assets.length > 0) { var outputValues = assets.map(function(item) { return [ item.symbol || "", // A欄 item.qty || 0, // B欄 item.price || 0, // C欄 item.currency || "", // D欄 item.type || "", // E欄 item.category || "", // F欄 "", "", "", "", "", "" // G~L 欄預留，若有欄位請自行補上對應 item 屬性 ]; }); // 一次性寫入試算表 sheet.getRange(2, 1, outputValues.length, 12).setValues(outputValues); } return ContentService.createTextOutput("Saved").setMimeType(ContentService.MimeType.TEXT); } } catch (err) { return ContentService.createTextOutput("Error: " + err.message).setMimeType(ContentService.MimeType.TEXT); } }
